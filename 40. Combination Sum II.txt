Given a collection of candidate numbers (candidates) and a target number (target), find all unique combinations in candidates where the candidate numbers sum to target.

Each number in candidates may only be used once in the combination.

Note: The solution set must not contain duplicate combinations.

 

Example 1:

Input: candidates = [10,1,2,7,6,1,5], target = 8
Output: 
[
[1,1,6],
[1,2,5],
[1,7],
[2,6]
]
Example 2:

Input: candidates = [2,5,2,1,2], target = 5
Output: 
[
[1,2,2],
[5]
]
 

/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum2 = function(candidates, target) {
    const result = [];
    // Sorting is mandatory here to handle duplicates and optimize
    candidates.sort((a, b) => a - b);

    function backtrack(remaining, start, path) {
        if (remaining === 0) {
            result.push([...path]);
            return;
        }

        for (let i = start; i < candidates.length; i++) {
            // Optimization: If current number is too big, no need to check further
            if (candidates[i] > remaining) break;

            // SKIP DUPLICATES: 
            // If this number is the same as the previous one in the same loop level, skip it.
            if (i > start && candidates[i] === candidates[i - 1]) continue;

            path.push(candidates[i]);
            
            // Note: We pass 'i + 1' to ensure each element is used only once.
            backtrack(remaining - candidates[i], i + 1, path);
            
            path.pop();
        }
    }

    backtrack(target, 0, []);
    return result;
};
